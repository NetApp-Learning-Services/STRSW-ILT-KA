1.  Install Go on Linux: 
    curl -LO https://dl.google.com/go/go1.14.4.linux-amd64.tar.gz
    tar -C /usr/local -xzf go1.14.4.linux-amd64.tar.gz
    export PATH=$PATH:/usr/local/go/bin

2. Download etcdctl:

[root@rhel3 ~]# go get -v go.etcd.io/etcd/etcdctl

3. Local the etcdctl and run it:

[root@rhel3 bin]# pwd
/root/go/bin

4. Identify the configuration of the etcd master pod by describing the pod:

    - --advertise-client-urls=https://192.168.0.63:2379
    - --cert-file=/etc/kubernetes/pki/etcd/server.crt
    - --client-cert-auth=true
    - --data-dir=/var/lib/etcd
    - --initial-advertise-peer-urls=https://192.168.0.63:2380
    - --initial-cluster=rhel3=https://192.168.0.63:2380
    - --key-file=/etc/kubernetes/pki/etcd/server.key
    - --listen-client-urls=https://127.0.0.1:2379,https://192.168.0.63:2379
    - --listen-metrics-urls=http://127.0.0.1:2381
    - --listen-peer-urls=https://192.168.0.63:2380
    - --name=rhel3
    - --peer-cert-file=/etc/kubernetes/pki/etcd/peer.crt
    - --peer-client-cert-auth=true
    - --peer-key-file=/etc/kubernetes/pki/etcd/peer.key
    - --peer-trusted-ca-file=/etc/kubernetes/pki/etcd/ca.crt
    - --snapshot-count=10000
    - --trusted-ca-file=/etc/kubernetes/pki/etcd/ca.crt

5. Get members within your etcd cluster:

[root@rhel3 bin]# ETCDCTL_API=3  ./etcdctl --endpoints=192.168.0.63:2379 --cacert /etc/kubernetes/pki/etcd/server.crt --cert /etc/kubernetes/pki/etcd/server.crt --key /etc/kubernetes/pki/etcd/server.key member list
91a7f0234ef57cfe, started, rhel3, https://192.168.0.63:2380, https://192.168.0.63:2379, false
